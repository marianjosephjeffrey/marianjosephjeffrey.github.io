#!/usr/bin/env python3
# -*- coding: utf-8 -*-

import xml.etree.ElementTree as ET
import argparse
import gzip
def parse_nessus(file_path):
    with open(file_path, "rb") as f:
        raw = f.read()

    # auto-detect gzip
    if raw[:2] == b"\x1f\x8b":
        raw = gzip.decompress(raw)

    root = ET.fromstring(raw)
from collections import defaultdict
from reportlab.platypus import SimpleDocTemplate, Paragraph, Spacer, PageBreak
from reportlab.lib.styles import getSampleStyleSheet
from reportlab.lib.pagesizes import letter
from reportlab.lib.units import inch


def parse_nessus(file_path):
    tree = ET.parse(file_path)
    root = tree.getroot()

    results = defaultdict(lambda: defaultdict(list))

    for host in root.iter("ReportHost"):
        host_name = host.get("name", "Unknown Host")

        for item in host.iter("ReportItem"):
            severity = int(item.get("severity", 0))
            plugin = item.get("pluginName", "Unknown Plugin")

            synopsis = item.findtext("synopsis", default="")
            solution = item.findtext("solution", default="")
            plugin_output = item.findtext("plugin_output", default="")
            cvss = item.findtext("cvss_base_score", default="")

            results[host_name][severity].append({
                "plugin": plugin,
                "synopsis": synopsis,
                "solution": solution,
                "output": plugin_output,
                "cvss": cvss
            })

    return results


def build_pdf(data, output_file):
    styles = getSampleStyleSheet()
    doc = SimpleDocTemplate(output_file, pagesize=letter)

    elements = []

    severity_names = {
        4: "Critical",
        3: "High",
        2: "Medium",
        1: "Low",
        0: "Info"
    }

    elements.append(Paragraph("<b>Nessus Scan Report</b>", styles["Title"]))
    elements.append(Spacer(1, 0.2 * inch))

    for host, severities in data.items():
        elements.append(Paragraph(f"<b>Host: {host}</b>", styles["Heading2"]))
        elements.append(Spacer(1, 0.1 * inch))

        for sev in sorted(severities.keys(), reverse=True):
            elements.append(
                Paragraph(f"<b>Severity: {severity_names.get(sev, sev)}</b>",
                          styles["Heading3"])
            )

            for vuln in severities[sev]:
                elements.append(Paragraph(f"<b>{vuln['plugin']}</b>", styles["BodyText"]))

                if vuln["cvss"]:
                    elements.append(Paragraph(f"CVSS: {vuln['cvss']}", styles["BodyText"]))

                if vuln["synopsis"]:
                    elements.append(
                        Paragraph(f"<i>Summary:</i> {vuln['synopsis']}", styles["BodyText"])
                    )

                if vuln["solution"]:
                    elements.append(
                        Paragraph(f"<i>Solution:</i> {vuln['solution']}", styles["BodyText"])
                    )

                if vuln["output"]:
                    short_output = vuln["output"][:1000]
                    elements.append(
                        Paragraph(f"<i>Plugin Output:</i><br/>{short_output}",
                                  styles["BodyText"])
                    )

                elements.append(Spacer(1, 0.15 * inch))

        elements.append(PageBreak())

    doc.build(elements)


def main():
    parser = argparse.ArgumentParser(
        description="Convert Nessus .nessus file to PDF (Python3)"
    )
    parser.add_argument("input", help="Input .nessus file")
    parser.add_argument("-o", "--output",
                        default="nessus_report.pdf",
                        help="Output PDF filename")

    args = parser.parse_args()

    data = parse_nessus(args.input)
    build_pdf(data, args.output)

    print(f"✔ PDF generated: {args.output}")


if __name__ == "__main__":
    main()
