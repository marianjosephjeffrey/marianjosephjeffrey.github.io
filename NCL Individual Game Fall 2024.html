import crc32c

# Function to calculate CRC32C checksum for an array of strings
def calculate_crc32c_checksums(messages):
    return [format(crc32c.crc32c(message.encode()), '08x') for message in messages]

# Example usage
messages = [
    "I just got home, finally!",
    "Let me know when you're on your way.",
    "Did you see the game last night?",
    # Add more messages here
]

checksums = calculate_crc32c_checksums(messages)
print(checksums)
